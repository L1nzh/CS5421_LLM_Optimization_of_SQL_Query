```sql
SELECT
  d.department_id,
  ec.employee_cnt
FROM (
  SELECT
    department_id,
    COUNT(*) AS employee_cnt
  FROM employees
  WHERE is_active = TRUE
  GROUP BY department_id
  HAVING COUNT(*) >= 10
) ec
INNER JOIN departments d ON ec.department_id = d.department_id;
```
