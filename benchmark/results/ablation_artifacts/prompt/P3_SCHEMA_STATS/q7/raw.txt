SELECT
  i.i_item_id,
  avg(ss.ss_quantity) agg1,
  avg(ss.ss_list_price) agg2,
  avg(ss.ss_coupon_amt) agg3,
  avg(ss.ss_sales_price) agg4
FROM store_sales ss
INNER JOIN date_dim d ON ss.ss_sold_date_sk = d.d_date_sk AND d.d_year = 2000
INNER JOIN customer_demographics cd ON ss.ss_cdemo_sk = cd.cd_demo_sk AND cd.cd_gender = 'M' AND cd.cd_marital_status = 'S' AND cd.cd_education_status = 'College'
INNER JOIN item i ON ss.ss_item_sk = i.i_item_sk
INNER JOIN promotion p ON ss.ss_promo_sk = p.p_promo_sk AND (p.p_channel_email = 'N' OR p.p_channel_event = 'N')
GROUP BY i.i_item_id
ORDER BY i.i_item_id
LIMIT 100;
