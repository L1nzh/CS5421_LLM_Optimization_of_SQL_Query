SELECT
  dt.d_year,
  item.i_brand_id brand_id,
  item.i_brand brand,
  SUM(ss.ss_ext_sales_price) sum_agg
FROM date_dim dt
INNER JOIN store_sales ss ON dt.d_date_sk = ss.ss_sold_date_sk
INNER JOIN item ON ss.ss_item_sk = item.i_item_sk
WHERE item.i_manufact_id = 128
  AND dt.d_moy = 11
GROUP BY dt.d_year, item.i_brand, item.i_brand_id
ORDER BY dt.d_year, sum_agg DESC, brand_id
LIMIT 100;
